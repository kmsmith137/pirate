# metadata files sent from pirate to the grouper. (Most fields are only useful internally,

ntrees: 10                     # Number of output trees (== length of the 'trees' sequence below)

# As explained in notes/tree_dedispersion.tex, the dedisperser consists of multiple "trees"
# corresponding to pairs (primary_tree_index, early_trigger_level). Here, primary_tree_index
# (denoted p) selects the primary tree: the input is time-downsampled by 2^p before dedispersion,
# which controls the DM-range of the tree. If early_trigger_level > 0, then the tree has an
# "early trigger" and searches a subset of the frequency range (the level is the "earliness").
# 
# The details of the trees are nontrivial -- see notes/tree_dedispersion.tex for info/plots.
